# Fetch Users
query fetchUsers ($order: String){
  users (order: $order){
    id
    name
    age
  }
}

# Fetch Posts
query fetchPosts{
  posts (query: "it"){
    id
    title
    body
    published
  }
}

# Fetch comments
query fetchComments{
  comments{
    id
    text
  }
}


# Fetching users -> posts -> comments
query fetchUsers{
  users{
    id
    name
    age
    posts{
      id
      title
      body
      published
      comments{
        id
        text
      }
    }
  }
}
